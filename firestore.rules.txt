rules_version = '2';
service cloud.firestore {
  match /databases/{database}/documents {

    // --- HELPER FUNCTIONS ---

    function isAuthenticated() {
      return request.auth != null;
    }

    function isUser(userId) {
      return isAuthenticated() && request.auth.uid == userId;
    }

    // Hardcoded Admin Check for MVP
    function isAdmin() {
      return isAuthenticated() && 
             request.auth.token.email == 'kayky@gmail.com';
    }

    // Verifica se o usuário tem um documento na subcoleção 'members' da banda
    function isMember(bandId) {
      return exists(/databases/$(database)/documents/bands/$(bandId)/members/$(request.auth.uid));
    }

    // Busca a role do usuário na banda com segurança (verifica existência antes)
    function getRole(bandId) {
      let memberPath = /databases/$(database)/documents/bands/$(bandId)/members/$(request.auth.uid);
      return exists(memberPath) ? get(memberPath).data.role : null;
    }

    function isLeader(bandId) {
      return getRole(bandId) == 'leader';
    }

    function isVice(bandId) {
      let role = getRole(bandId);
      return role == 'vice' || role == 'leader';
    }
    
    function canEditMusic(bandId) {
       let role = getRole(bandId);
       return role == 'leader' || role == 'vice' || role == 'regente';
    }

    function canEditBand(bandId) {
       let role = getRole(bandId);
       return role == 'leader' || role == 'vice' || role == 'regente';
    }

    // --- COLLECTION RULES ---

    // 1. USERS
    match /users/{userId} {
      allow read: if isUser(userId) || isAdmin();
      allow write: if isUser(userId) || isAdmin();
    }

    // 2. BANDS
    match /bands/{bandId} {
      // Create: Qualquer usuário autenticado pode criar uma banda
      allow create: if isAuthenticated();
      
      // Read: Necessário para buscar a banda pelo código antes de entrar
      allow read: if isAuthenticated();

      // Update:
      // - Cenário A: Usuário entrando/saindo da banda (atualiza APENAS memberCount)
      // - Cenário B: Líder, Vice ou Regente editando dados da banda
      // - Cenário C: Admin atualizando status da assinatura
      allow update: if (isAuthenticated() && request.resource.data.diff(resource.data).affectedKeys().hasOnly(['memberCount'])) 
                    || canEditBand(bandId)
                    || isAdmin();
      
      // Delete: Apenas o Líder ou Admin
      allow delete: if isLeader(bandId) || isAdmin();

      // 2.1 MEMBERS (Subcoleção)
      match /members/{memberId} {
        allow read: if isUser(memberId) || isMember(bandId) || isAdmin();
        allow write: if (isUser(memberId) && request.resource.data.role == 'musico') 
                     || isVice(bandId)
                     || isAdmin();
      }

      // 2.2 SONGS (Subcoleção)
      match /songs/{songId} {
        allow read: if isMember(bandId);
        allow write: if canEditMusic(bandId);
      }

      // 2.3 PLAYLISTS (Subcoleção)
      match /playlists/{playlistId} {
        allow read: if isMember(bandId);
        allow write: if canEditMusic(bandId);
      }

      // 2.4 REGENCY SESSION (Subcoleção)
      match /regency/session {
        allow read: if isMember(bandId);
        allow write: if canEditMusic(bandId);
      }

      // 2.5 BILLING (Subcoleção - Deprecated but kept for legacy)
      match /billing/{document=**} {
        allow read: if isMember(bandId) || isAdmin();
        allow write: if canEditBand(bandId) || isAuthenticated() || isAdmin();
      }

      match /billingLogs/{logId} {
        allow read: if canEditBand(bandId) || isAdmin();
        allow write: if canEditBand(bandId) || isAdmin();
      }
    }

    // 3. SUBSCRIPTIONS (New Collection)
    match /subscriptions/{subId} {
      // Read: Owner of the subscription (via ownerId field) or Admin
      allow read: if (isAuthenticated() && resource.data.ownerId == request.auth.uid) || isAdmin();
      // Write: Only Admin (Manual process) or System (via Cloud Functions if implemented later)
      // For now, create is allowed for authenticated users ONLY during band creation (handled by batch write in client code, but rules check individual ops)
      // Actually, since we do client-side batch, the user needs create permission.
      allow create: if isAuthenticated(); 
      allow update: if isAdmin();
    }

    // 4. MANUAL PAYMENTS (New Collection)
    match /manualPayments/{paymentId} {
      allow read: if isAdmin();
      allow write: if isAdmin();
    }
  }
}
